// Motor Shield Test Code
#include <AFMotor.h>  // Include library for motor shield

// Create motor objects for M1 and M2
AF_DCMotor motor1(1);  // M1 is motor 1
AF_DCMotor motor2(2);  // M2 is motor 2

void setup() {
  // Set motor speed to 255 (full speed)
  motor1.setSpeed(255);  // Set speed for motor 1
  motor2.setSpeed(255);  // Set speed for motor 2
}

void loop() {
  // Move motor 1 forward
  motor1.run(FORWARD);
  motor2.run(FORWARD);
  delay(2000);  // Run motors for 2 seconds

  // Stop motor 1
  motor1.run(RELEASE);
  motor2.run(RELEASE);
  delay(1000);  // Wait for 1 second

  // Move motor 1 forward again
  motor1.run(FORWARD);
  motor2.run(FORWARD);
  delay(2000);  // Run motors for 2 seconds

  // Stop motor 1
  motor1.run(RELEASE);
  motor2.run(RELEASE);
  delay(1000);  // Wait for 1 second
}
